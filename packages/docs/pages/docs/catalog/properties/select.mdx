import '@composify/react/style.css';
import { PropertyControlSelect } from '@composify/react/editor';
import { useState } from 'react';

export const BasicUsage = () => {
  const [value, setValue] = useState('apple');

  return (
    <PropertyControlSelect
      name="example"
      spec={{
        label: 'Fruit',
        type: 'select',
        options: [
          { label: 'Apple', value: 'apple' },
          { label: 'Banana', value: 'banana' },
          { label: 'Orange', value: 'orange' },
        ],
      }}
      value={value}
      onChange={(_, next) => setValue(next ?? 'apple')}
    />
  );
};

# Select Property

The **select** property renders a dropdown, letting users pick one option from a predefined list.
It's great for choosing from categories, sizes, or any fixed set of values. 

:::code-group
<div data-title="Preview">
  <BasicUsage />
</div>

```tsx showLineNumbers [Code]
Catalog.register('Sample', {
  component: Sample,
  props: {
    fruit: {
      label: 'Fruit',
      type: 'select',// [!code hl]
      options: [// [!code hl]
        { label: 'Apple', value: 'apple' },// [!code hl]
        { label: 'Banana', value: 'banana' },// [!code hl]
        { label: 'Orange', value: 'orange' },// [!code hl]
      ],// [!code hl]
    },
  },
});
```
:::

## Required Parameters

### `options`

The `options` property is an array that defines the choices for the dropdown.
Each option in the array should be an object with: 

- `label`: The text users will see in the list.
- `value`: The actual value that gets passed to your component.

```tsx showLineNumbers
Catalog.register('Sample', {
  component: Sample,
  props: {
    fruit: {
      label: 'Fruit',
      type: 'select',
      options: [// [!code hl]
        { label: 'Apple', value: 'apple' },// [!code hl]
        { label: 'Banana', value: 'banana' },// [!code hl]
        { label: 'Orange', value: 'orange' },// [!code hl]
      ],// [!code hl]
    },
  },
});
```
