import '@composify/react/style.css';
import { PropertyControlNumber } from '@composify/react/editor';
import { useState } from 'react';

export const BasicUsage = () => {
  const [value, setValue] = useState(0);

  return (
    <PropertyControlNumber
      name="example"
      spec={{
        label: 'Count',
        type: 'number',
      }}
      value={value}
      onChange={(_, next) => setValue(next ?? 0)}
    />
  );
};

# Number Property

A **number** property holds numeric values, perfect for things like counts, sizes, spacing, or any other prop that needs a number.
