import '@composify/react/style.css';
import { PropertyControlBoolean } from '@composify/react/editor';
import { useState } from 'react';

export const BasicUsage = () => {
  const [value, setValue] = useState(false);

  return (
    <PropertyControlBoolean
      name="example"
      spec={{
        label: 'Visible',
        type: 'boolean',
      }}
      value={value}
      onChange={(_, next) => setValue(next)}
    />
  );
};

# Boolean Property

A **boolean** property is a simple `true`/`false` toggle, perfect for things like showing or hiding a component, enabling a feature, or any other on/off switch.
