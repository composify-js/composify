import '@composify/react/style.css';
import { PropertyControlText } from '@composify/react/editor';
import { useState } from 'react';

export const BasicUsage = () => {
  const [value, setValue] = useState('');

  return (
    <PropertyControlText
      name="example"
      spec={{
        type: 'text',
        label: 'Title',
        placeholder: 'Type something...',
      }}
      value={value}
      onChange={(_, next) => setValue(next ?? '')}
    />
  );
};

# Text property

A text prop is just a plain string. Use it for titles, labels, names, or any short bit of text content.

:::code-group

</div>

```tsx showLineNumbers [Code]
Catalog.register('Sample', {

    title: {
      label: 'Title',

    },
  },
});
```

## Optional Parameters

### `placeholder`

Placeholder lets you show hint text in the input when it's empty.

