import '@composify/react/style.css';
import { PropertyControlTextArea } from '@composify/react/editor';
import { useState } from 'react';

export const BasicUsage = () => {
  const [value, setValue] = useState('');

  return (
    <PropertyControlTextArea
      name="example"
      spec={{
        type: 'textarea',
        label: 'Description',
        placeholder: 'Type something...',
      }}
      value={value}
      onChange={(_, next) => setValue(next ?? '')}
    />
  );
};

# Textarea property

A textarea prop is for multi-line strings. Use it when you need longer input, like descriptions, messages, or paragraphs of text.

:::code-group
<div data-title="Preview">
  <BasicUsage />
</div>

```tsx showLineNumbers [Code]
Catalog.register('Sample', {
  component: Sample,
  props: {
    description: {
      label: 'Description',
      type: 'textarea',// [!code hl]
      placeholder: 'Type something...',// [!code hl]
    },
  },
});
```
:::

## Optional Parameters

### `placeholder`

Placeholder lets you show hint text in the input when it's empty.

```tsx showLineNumbers

    },
  },
});
```
